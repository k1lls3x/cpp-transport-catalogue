cmake_minimum_required(VERSION 3.10)
project(transport_catalogue VERSION 1.0 LANGUAGES CXX)

# Включаем C++20 без расширений
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Исполняемый target со всеми исходниками
add_executable(transport_catalogue
    main.cpp
    geo.cpp
    json.cpp
    json_builder.cpp
    json_reader.cpp
    map_renderer.cpp
    svg.cpp
    transport_catalogue.cpp
    transport_router.cpp
)

# Инклюдим корневую папку для хэдеров
target_include_directories(transport_catalogue
    PRIVATE
        ${CMAKE_SOURCE_DIR}
)

# Включаем «максимальные» предупреждения
if (MSVC)
    target_compile_options(transport_catalogue PRIVATE /W4 /permissive-)
else()
    target_compile_options(transport_catalogue PRIVATE -Wall -Wextra -pedantic)
endif()
